-- количество игровых сессий 
select    date_trunc('month', start_session) as mm
				, count(*) as cnt_session_all
      , sum(case when end_session - start_session > interval '5 minute' 
                 then 1.0 else 0.0 end) as cnt_session_signif
			, sum(case when end_session - start_session > interval '5 minute' 
                 then 1.0 else 0.0 end) / count(*) as share_signif
from skygame.game_sessions
group by mm
order by mm 

-- количество игровых сессий более 5 мин

select  date_trunc('month', start_session) as mm
				, avg(end_session - start_session) as avg_len
from skygame.game_sessions
where end_session - start_session > interval '5 minute'
group by mm
order by mm

-- количество игровых сессий длинее 1 часа

				, sum(case when end_session - start_session > interval '1 hour' 
									 then 1.0 else 0.0 end) / count(*) as share_long
from skygame.game_sessions
where end_session - start_session > interval '5 minute'
group by mm
order by mm

select    date_trunc('month',dtime_pay) as mm
        , type
	      , sum(cnt_buy * price) as revenue
from skygame.monetary m
   join skygame.item_list i
      on m.id_item_buy = i.id_item
   join skygame.log_prices p
		  on m.id_item_buy = p.id_item
      and m.dtime_pay >= p.valid_from
      and m.dtime_pay <= coalesce(valid_to, to_date('01/01/3000', 'DD/MM/YYYY'))
group by   mm
         , type
order by mm

				, avg(cnt_buy) as cnt
	      , sum(cnt_buy * price) as revenue

	   join skygame.item_list i

select  *
      ,extract ('day' from ((select max (dtime_pay) from skygame.monetary) - mm )/30) as interv
      ,avg_rev/( extract ('day' from ((select max (dtime_pay) from skygame.monetary) - mm )/30)) avg_rev_per_month
from (
select    date_trunc('month', reg_date) as mm
        , sum(cnt_buy * price) as revenue
        , count(distinct m.id_user) as cnt
        , sum(cnt_buy * price) / count(distinct m.id_user) as avg_rev
from skygame.monetary m
   join skygame.log_prices p
	  on m.id_item_buy = p.id_item
      and m.dtime_pay >= p.valid_from
      and m.dtime_pay <= coalesce(valid_to, to_date('01/01/3000', 'DD/MM/YYYY'))
   join skygame.users u
      on m.id_user = u.id_user
where reg_date < (select max(dtime_pay) - interval '1 month' from skygame.monetary)
group by mm
order by mm

-- считаем количество месяцев от момента регистрации до последней игры
lt_mm as (
select
su.id_user
, ceil(extract ('day' from max(start_session) - min(reg_date))/30) as lt_mm
from skygame.users su
join skygame.game_sessions gs
on su.id_user = gs.id_user
group by su.id_user
)
--джойним два CTE, считаем среднюю выручку за месяц жизни
select ltr.id_user
, revenue/lt_mm as ltr_mm
from ltr
join lt_mm
on ltr.id_user = lt_mm.id_user
order by ltr_mm desc
